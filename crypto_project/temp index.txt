    <script>
        // Create floating particles
        function createParticles() {
            const container = document.getElementById('particles-container');
            for (let i = 0; i < 25; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.animationDelay = Math.random() * 15 + 's';
                particle.style.animationDuration = (12 + Math.random() * 8) + 's';
                container.appendChild(particle);
            }
        }
        createParticles();

        // Main encryption execution
        document.getElementById('run-button').addEventListener('click', async () => {
            const plaintext = document.getElementById('plaintext').value;
            const resultsDiv = document.getElementById('results');
            const loadingDiv = document.getElementById('loading');
            const runButton = document.getElementById('run-button');
            
            // Validation
            if (!plaintext.trim()) {
                alert('⚠️ Please enter a message to encrypt!');
                return;
            }

            // Hide results and show loading
            resultsDiv.classList.add('hidden');
            loadingDiv.classList.remove('hidden');
            runButton.disabled = true;
            runButton.querySelector('span').textContent = 'Processing...';

            try {
                const response = await fetch('/run_crypto', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ plaintext: plaintext }),
                });

                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }

                const data = await response.json();
                
                // Check for error status
                if (data.status === 'Error') {
                    throw new Error(data.message || 'Unknown error occurred');
                }
                
                // Populate Kyber results
                document.getElementById('kyber-pk').textContent = data.kyber_pk || 'N/A';
                document.getElementById('kyber-sk').textContent = data.kyber_sk || 'N/A';
                document.getElementById('enc-secret').textContent = data.encapsulated_secret || 'N/A';
                document.getElementById('dec-secret').textContent = data.decapsulated_secret || 'N/A';
                
                // Update secrets match status
                const matchSpan = document.getElementById('secrets-match');
                const matchIcon = document.getElementById('match-icon');
                
                if (data.secrets_match) {
                    matchSpan.textContent = '✓ VERIFIED';
                    matchSpan.className = 'text-3xl font-bold status-success';
                    matchIcon.className = 'w-8 h-8 status-success';
                } else {
                    matchSpan.textContent = '✗ MISMATCH';
                    matchSpan.className = 'text-3xl font-bold status-fail';
                    matchIcon.className = 'w-8 h-8 status-fail';
                }
                
                // Populate ChaCha20 results
                document.getElementById('original-text').textContent = data.original_plaintext || 'N/A';
                document.getElementById('chacha-cipher').textContent = data.chacha_ciphertext || 'N/A';
                document.getElementById('decrypted-text').textContent = data.decrypted_plaintext || 'N/A';

                // Hide loading and show results with animation
                loadingDiv.classList.add('hidden');
                resultsDiv.classList.remove('hidden');
                
                // Scroll to results smoothly
                setTimeout(() => {
                    resultsDiv.scrollIntoView({ behavior: 'smooth', block: 'start' });
                }, 100);

            } catch (error) {
                console.error('Error:', error);
                loadingDiv.classList.add('hidden');
                
                // Show error message
                const errorHtml = `
                    <div class="glass rounded-2xl p-8 text-center border-2 border-red-500/30">
                        <svg class="w-16 h-16 text-red-400 mx-auto mb-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                            <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M12 8v4m0 4h.01M21 12a9 9 0 11-18 0 9 9 0 0118 0z"></path>
                        </svg>
                        <h3 class="text-2xl font-bold text-red-300 mb-2">Connection Error</h3>
                        <p class="text-gray-400 mb-4">Unable to reach the cryptography backend server.</p>
                        <p class="text-sm text-gray-500">
                            <strong>Possible causes:</strong><br>
                            • Flask server is not running<br>
                            • Server is running on a different port<br>
                            • Network connectivity issues
                        </p>
                        <div class="mt-6 p-4 bg-slate-900/50 rounded-lg text-left">
                            <p class="text-xs text-gray-400 font-mono">Error details: ${error.message}</p>
                        </div>
                        <p class="text-sm text-purple-300 mt-4">
                            💡 Tip: Run <code class="bg-slate-800 px-2 py-1 rounded">python app.py</code> to start the server
                        </p>
                    </div>
                `;
                
                resultsDiv.innerHTML = errorHtml;
                resultsDiv.classList.remove('hidden');
            } finally {
                runButton.disabled = false;
                runButton.querySelector('span').textContent = 'Execute Cryptographic Protocol';
            }
        });

        // Allow Enter key to submit (with Ctrl/Cmd)
        document.getElementById('plaintext').addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && (e.ctrlKey || e.metaKey)) {
                e.preventDefault();
                document.getElementById('run-button').click();
            }
        });

        // Add visual feedback on textarea focus
        const textarea = document.getElementById('plaintext');
        textarea.addEventListener('focus', () => {
            textarea.parentElement.classList.add('ring-2', 'ring-purple-500/50');
        });
        textarea.addEventListener('blur', () => {
            textarea.parentElement.classList.remove('ring-2', 'ring-purple-500/50');
        });
    </script>
